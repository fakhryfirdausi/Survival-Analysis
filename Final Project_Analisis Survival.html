def prio_count(x):
    if x >5:
        return '>5 convictions'
    else:
        return '<5 convictions'
def age_groups(x):
    if x<30:
        return 'less than 30'
    else:
        return 'more than 30'
    
    
df['prio_cat'] = df['prio'].apply(lambda x:prio_count(x))
df['new_age_group'] = df['age'].apply(lambda x:age_groups(x))

In [64]:
### Check for multicollinearity as it is a basic assumption of the CPH model
from statsmodels.stats.outliers_influence import variance_inflation_factor 
  
# creating dummies for gender   
# the independent variables set 
X = df[['week', 'arrest', 'fin', 'age', 'race', 'wexp', 'mar', 'paro', 'prio']] 
  
# VIF dataframe 
vif_data = pd.DataFrame() 
vif_data["feature"] = X.columns 
  
# calculating VIF for each feature 
vif_data["VIF"] = [variance_inflation_factor(X.values, i) 
                          for i in range(len(X.columns))] 
  
print(vif_data)

kmf = KaplanMeierFitter()

# Fitting KMF to our data
kmf.fit(durations = df['week'], event_observed = df['arrest'],label = 'Estimate for released prisoners')

# plotting the KMF curve
fig,ax = plt.subplots(figsize=(10,8))
kmf.plot(at_risk_counts = True,ax=ax)
ax.set_title('Kaplan-Meier Survival Curve — All Released Prisoners')
ax.set_xlabel('Released (weeks)')
ax.set_ylabel('Re-Arrested Probability')
ax.grid();

ax = kmf.plot()
